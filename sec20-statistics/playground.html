def mode(arr):
    values, counts = np.unique(arr, return_counts=True)
    modes = values[counts == counts.max()]
    
    return np.array(modes)

## testing the bessel correction

import numpy as np

np.random.seed(42)

population_size = 10_000_000
population = np.random.normal(0, 10, population_size)

true_population_variance = np.var(population, ddof=0)

## ddof --> delta degree of freedom 
    ## 1 -> Population Variance
    ## 0 --> Sample Variance
    
    
sample_size = 1000
iterations = 10_000

sample_variances_n = np.empty(iterations)

for i in range(iterations):
    sample = np.random.choice(population, size=sample_size, replace=True)
    sample_variances_n[i] = np.var(sample, ddof=0)

average_sample_variance_n = sample_variances_n.mean()

print("True Population Variance       :", true_population_variance)
print("Average Sample Variance (n)    :", average_sample_variance_n)
print("Bias                           :", average_sample_variance_n - true_population_variance)

arr = np.array([1,2,2,2,3,3,4,5,5,5,6,6,6,6,7,8,8,9,27])
## 27 can be considered as on outlier 


## Fence range of data

##     [Lower Fence <------------------------> Higher Fence]     ## --> values outside this range are considered as outliers

'''
## Lower Fence = 1st Quartile - 1.5 * IQR
## Higher Fence = 3rd Quartile + 1.5 * IQR

## IQR = 3rd Quartile - 1st Quartile
'''



lowFence = np.percentile(arr, 25) - 1.5 * (np.percentile(arr, 75) - np.percentile(arr, 25))
highFence = np.percentile(arr, 75) + 1.5 * (np.percentile(arr, 75) - np.percentile(arr, 25))

print(f"Lower Fence : {lowFence}")
print(f"Higher Fence : {highFence}")


## Outliers 

outliers = np.array([num for num in arr if num < lowFence or num > highFence])
print(f"Outliers are {outliers}")


## array cleaned (Basically outliers are removed)
arr = [num for num in arr if num >= lowFence and num <= highFence]


## 5 numbers values

print(f"Min is {np.min(arr)}")
print(f"1st Quartile is {np.percentile(arr, 25)}")
print(f"Median is {np.median(arr)}")
print(f"3rd Quartile is {np.percentile(arr, 75)}")
print(f"Max is {np.max(arr)}")

## Advantage of Covariance
    ## We can quantify the relationship between two variables
    

## Disadvantage of Covariance
    ## The covariance value is not limited 
            ## This means that the value can go from -inf to +inf
            
            ## A, B, C  --> Three random variables 
            ## Cov(A, B) -> 30
            ## Cov(B, C) -> 40
            ## Cov(A, C) -> 50
            
            ## So we cannot say that Cov(A, B) > Cov(B, C) since there is no limit  
                ## the covariance can also be +300 or -100
                
                
                ## basically we cannot compare two covariance values



## Solution to this problem is Correlation Coefficient

## The pearson correlation coefficient limits the value to ********[-1 to 1]************

## Formula 

'''
        ## r = Cov(x, y) / (std(x) * std(y))   
                                ## r --> correlation coefficient (pearson)

'''

## Because of this, we can say which is stronger 

## The more value towards the 1, it is more positively correlated 
## The more value towards the -1, it is more negatively correlated



## Example:
HoursStudy = np.array([2,3,4,5,6])
Marks = np.array([50,60,70,80, 90])

covariance = np.cov(HoursStudy, Marks, ddof = 1)[0][1]


## ddof --> degree of freedom 
        ## 1 --> sample covariance
        ## 0 --> population covariance
        
        
pearsonCorrelationCoefficient = covariance / (np.std(HoursStudy, ddof= 1) * np.std(Marks, ddof= 1))
print(f"The value of pearson correlation coefficient is {pearsonCorrelationCoefficient}") 
## The value of pearson correlation coefficient is 1.0

## the value is 1 , because there is no noise in the data

The value of pearson correlation coefficient is 1.0

## Formula of Spearman Correlation Coefficient

'''
            ## r_s = Cov(R(x), R(y)) / (std(R(x)) * std(R(y)))
                    ## R(x) and R(y) are the rank of x and y respectively
                    ## r_s = spearman correlation coefficient
'''

## How to find rank of a number

    ## The rank of a number is the number of numbers that are smaller than itself

'''
X    Y    R(X)    R(Y)
------------------------
2    50    1    1     
3    60    2    2
4    70    3    3
5    80    4    4        ## The rank is the actually the rank of the number in its own array 
6    90    5    5


'''

## Example:

X = np.array([2,3,4,5,6])
Y = np.array([50,60,70,80, 90])


rankX = np.argsort(X) + 1     ## +1 is added because the rank starts from 1
rankY = np.argsort(Y) + 1     ## argsort return the index of the sorted array

print(rankX)
print(rankY)

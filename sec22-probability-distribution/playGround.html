## This is the PMF constructed for the given example

'''
            PMF = ( p ^ k ) * ( (1 - p) ^ (1 - k) )
'''

## Mean of the Bernoulli Distribution 

    ## In order to calculate the mean of the distribution, we need to calculate the expected value of the random variable
    
    
'''
                E(x) = sigma( k = 0 to 1)  [   k * p(k)  ]   
                
                                                E(x) --> Expected Value
'''

## Example : 
            ## p = 0.6 
            ## E(x) = 0 * 0.4 + 1 * 0.6 = 0.6
            
## Expected Value = 0.6
        ## therefor the mean is 0.6, i.e. nothing but the value of p


## Median of the Bernouli Distribution 

'''
                Median = 0 if p < 0.5
                       = [0 , 1] if p = 0.5
                       = 1 if p > 0.5
                       
A better way to write it as

                Median = 0,  if  p < q
                        = 0.5 ,  if p = q
                        = 1,  if p > q
'''


## Mode for the bernouli distribution

'''
                if p > q, then mode = p
                else mode = q
'''


## Variance of the Bernouli Distribution

        ## Consider k = {0, 1},  P(X = 1) = p = 0.6 and P(X = 0) = q = 0.4

        ## now, var = ( 0 - 0.6)^2 * 0.4 + (1 - 0.6)^2 * 0.6
                    # = 0.24 (This is nothing but p * q)
            
            
'''
                sigma^2  = p * q
'''

2. Binomial Distribution

In[16]:
##  Discrete Random Variable 

## Each outcome has a binary Value 

## The experiments are done in n trials 

## Basically, a series of Bernouli trials is considered as Binomial Distribution



## Example: Tossing a coin 10 times
                ## n = 10


## Notation: 
'''
                Denoted by -->    B(n, p)
                
                n = number of trials
                p = probability of success [0,1]


        Successes, k belongs to {0,1,2,3,4,5,6,7,8 ..... n}
        
'''


## PMF 
'''

            P(n, k, p) = ( n combitorial k ) * ( p ^ k ) * ( (1 - p) ^ (n - k) )
            
            n = number of trials
            k = number of successes
            p = probability of success

'''

## Exmple: 


p = 0.6
n = 10
k = 5

def nCr(n, r):
    return math.factorial(n) // math.factorial(r) // math.factorial(n - r)

y = nCr(n, k) * (p ** k) * ((1 - p) ** (n - k))

print(y)  ## --> PMF for k = 5


## Graph is 
        ## X - axis --> k (No. of success)
        ## Y - axis --> PMF
        

p = 0.6
n = 20

y = np.array([nCr(n, k) * (p ** k) * ((1 - p) ** (n - k)) for k in range(n + 1)])

print(y)

plt.figure(figsize=(5, 3))
sns.scatterplot(x=np.arange(n + 1), y=y)
plt.xlabel("Number of Successes")

plt.title("Binomial Distribution")

## Example: Quality Control

n = 10; p = 0.1
k = np.arange(0, 11)

def binomialDistribution(n, k, p):
    ans = nCr(n, k) * (p ** k) * ((1 - p) ** (n - k))

## One important parameter is lamda 


'''
            lamda = expected number of events occuring in the interval 
'''

## PMF 
'''
            PMF = P(X) =   ( e ^ (-lamda)  * lamda ^ x  ) / x!
            
            lamda = expected number of events occuring in the interval 
            x = random varaible 
            

'''

## Exmaple, for lamda = 3, what is the probability of a person visiting the hospital in 5th hour?

def poissonDistribution(lamda, x):
    return ((lamda ** x)  * math.exp(-lamda)) / math.factorial(x)

ans = poissonDistribution(lamda= 3, x = 5)
print(ans)
